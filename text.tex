\section*{Энтропия Реньи}

\[H=\cfrac{1}{1-\alpha} \log \sum\limits_{i=1}^n p_i^{\alpha} \ \text{ или } \ \cfrac{1}{1-\alpha} \log \int\limits_{-\infty}^{+\infty} p_i(x)^{\alpha}dx \]

\begin{itemize}
    \item $\alpha=0 \Rightarrow H=\log n$
    \item $\alpha\to 1 \Rightarrow H=-\sum\limits_{i=1}^n p_i\log p_i$
    \item $\alpha=2 \Rightarrow H=-\log \sum\limits_{i=1}^n p_i^2 $
\end{itemize}

\section*{Расстояние Кульбака-Лейблера}

\[D(P\, ||\, Q)=\int\limits_{-\infty}^{+\infty} p(x)\log \left(\cfrac{p(x)}{q(x)}\right)dx \]

$\triangleright$ Обобщение Реньи:

\[D_{\alpha}(P\, ||\, Q)=\cfrac{1}{1-\alpha} \log \int\limits_{-\infty}^{+\infty} p(\mu)^{\alpha}q(\mu)^{1-\alpha}d\mu \]

\section*{Кросс-энтропия}

\[CH(p, q)=H(p)+D_{KL}(p\, || \, q)=-\int\limits_{-\infty}^{+\infty}p(x)\log q(x) dx  \]
